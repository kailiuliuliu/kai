副注、 sqlite常用命令

#sqlite3 test_db   --进入所需数据库/创建数据库

>.databases        --查看当前数据库
>.tables           --查看当前数据库中的表
>.help             --sqlite3帮助
>.schema           --各个表的生成语句
>sqlite3 testDB.db .dump > testDB.sql  --命令.dump导出数据库
>sqlite3 testDB.db < testDB.sql  --恢复数据库
.quit
