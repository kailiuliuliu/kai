一、 常用

git clone http://dev.kidf.eu:88/scroll/scroll-lang-cn-bundle.git

git status

git checkout

git add .
git add scroll-lang-cn-bundle/src/test

git checkout

git checkout -b develop //新建并切换到新分支

git commit

git checkout develop //切换到新分支
git push origin develop

git branch //查看所有本地分支
git branch -a //查看所有分支
git branch issue1 //新建一个分支，没有切换到新分支
//删除分支
git branch -d testbranch1
git push origin --delete testbranch1
git push

二、 回退

git log //先找到需要到达的哈希值

git reset --hard 2f86f9a823c744febdc79cca0616f6f93f31a14d //重置于此

三、 添加ssh key

1. 进入git bash

    git config --global user.name "kai"
    git config --global user.email "fengk1992@126.com"

    ssh-keygen -t rsa -C "fengk1992@126.com"
    //按3个回车，则密码为空。默认地址C:\Users\milk\.ssh

2. 在网站上添加id_rsa.pub里的内容
