二、 用户设定

SET PASSWORD FOR 'root'@'localhost' = PASSWORD('1');

CREATE USER 'milk'@'localhost' IDENTIFIED BY '1';

GRANT ALL ON helloontology.* TO 'milk'@'localhost';

CREATE DATABASE IF NOT EXISTS helloontology DEFAULT CHARSET utf8 COLLATE utf8_general_ci;

user pwd
root
user1 user1

grant select,insert,update,delete on *.* to user1@localhost Identified by "user1";

mysqld --initialize-insecure --user=user1

mysqld --initialize-insecure
